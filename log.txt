sepColumn <- function(v) {
  df <- data.frame(human_symbol=character(),
                   human_id=integer(),
                   mouse_id=integer(),
                   mouse_symbol=character(),
                   mgi=character(),
                   human_pheno=character())
  if (length(v[6]) > 1) {
    u <- strsplit(v[6], split = " ")
    for (mp in u) {
      newrow <- c(v[1:5], mp)
      df <- rbind(df,newrow)
    }
  }
  else {
    newrow <- c(v[1:5])
    df <- rbind(df,newrow)
  }
  return(df)
}

for (r in hmd_hp) {
  sepColumn(r)
}
